\subsection{Regular Expressions}

\subsubsection{Identifier}

\begin{verbatim}
[A-Za-z][0-9A-Za-z_\-]{0,15}
\end{verbatim}


\subsubsection{Image}

\begin{verbatim}
s?9(\([1-9][0-9]*\))?(v9(\([1-9][0-9]*\))?)?
\end{verbatim}


\subsubsection{Integer}

\begin{verbatim}
(\+|-)?([1-9][0-9]*)|(0)
\end{verbatim}


\subsubsection{Real}

\begin{verbatim}
(integer)(\.[0-9]+)
\end{verbatim}


\subsubsection{String}

\begin{verbatim}
'[0-9A-Za-z\+\-\*/:!\? ]*'
\end{verbatim}


\subsubsection{Comment}

\begin{verbatim}
(\*|/).*\.\n;
\end{verbatim}
